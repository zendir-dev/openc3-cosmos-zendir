  VIRTUAL
  ACCESSOR patch_xml_accessor.py
  APPEND_ITEM TIME 64 FLOAT
    KEY "/ElectromagneticSensor/Time/text()"
  APPEND_ITEM UTC 1024 STRING
    KEY "/ElectromagneticSensor/UTC/text()"
  APPEND_ITEM COUNT 64 FLOAT
    KEY "/ElectromagneticSensor/Data/Count/text()"
  APPEND_ITEM FREQUENCY 64 FLOAT
    KEY "/ElectromagneticSensor/Data/Frequency/text()"
  APPEND_ITEM GAIN 64 FLOAT
    KEY "/ElectromagneticSensor/Data/Gain/text()"

  APPEND_ITEM CCSDSVER      3 UINT   "CCSDS packet version number (See CCSDS 133.0-B-1)"
  APPEND_ITEM CCSDSTYPE     1 UINT   "CCSDS packet type (command or telemetry)"
    STATE TLM 0
    STATE CMD 1
  APPEND_ITEM CCSDSSHF      1 UINT   "CCSDS secondary header flag"
    STATE FALSE 0
    STATE TRUE 1
  APPEND_ID_ITEM CCSDSAPID 11 UINT 132 "CCSDS application process id"
  APPEND_ITEM CCSDSSEQFLAGS 2 UINT   "CCSDS sequence flags"
    STATE FIRST 1
    STATE CONT 0
    STATE LAST 2
    STATE NOGROUP 3
  APPEND_ITEM CCSDSSEQCNT  14 UINT   "CCSDS packet sequence count"
    OVERFLOW TRUNCATE
  APPEND_ITEM CCSDSLENGTH  16 UINT   "CCSDS packet data length"
  APPEND_ITEM XML 0 BLOCK "Electromagnetic Sensor XML Data"
    READ_CONVERSION hash_read_conversion.py TLM ZENDIR ELECTROMAGNETIC_SENSOR_XML
  ITEM TIME 0 0 DERIVED "Sensor timestamp (seconds)"
    READ_CONVERSION openc3/conversions/processor_conversion.py XMLPROCESSOR TIME
  ITEM UTC 0 0 DERIVED "UTC timestamp of the EM sensor data"
    READ_CONVERSION openc3/conversions/processor_conversion.py XMLPROCESSOR UTC
  ITEM COUNT 0 0 DERIVED "Detected device count"
    READ_CONVERSION openc3/conversions/processor_conversion.py XMLPROCESSOR COUNT
  ITEM FREQUENCY 0 0 DERIVED "Average detected frequency (Hz)"
    READ_CONVERSION openc3/conversions/processor_conversion.py XMLPROCESSOR FREQUENCY
  ITEM GAIN 0 0 DERIVED "Sensor gain"
    READ_CONVERSION openc3/conversions/processor_conversion.py XMLPROCESSOR GAIN
  PROCESSOR XMLPROCESSOR hash_processor.py XML
