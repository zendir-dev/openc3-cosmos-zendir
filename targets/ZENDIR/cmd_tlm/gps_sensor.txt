# ---------------------------------------------------------
# GPS Sensor Message (GPS_SENSOR) - APID 133
# ---------------------------------------------------------

TELEMETRY ZENDIR GPS_SENSOR_XML BIG_ENDIAN "GPS Sensor XML Structure"
  VIRTUAL
  ACCESSOR patch_xml_accessor.py
  APPEND_ITEM TIME 64 FLOAT
    KEY "/GPSSensor/Time/text()"
  APPEND_ITEM UTC 1024 STRING
    KEY "/GPSSensor/UTC/text()"
  APPEND_ITEM LATITUDE 64 FLOAT
    KEY "/GPSSensor/Data/Latitude/text()"
  APPEND_ITEM LONGITUDE 64 FLOAT
    KEY "/GPSSensor/Data/Longitude/text()"
  APPEND_ITEM ALTITUDE 64 FLOAT
    KEY "/GPSSensor/Data/Altitude/text()"

TELEMETRY ZENDIR GPS_SENSOR BIG_ENDIAN "Packet that contains GPS sensor data."
  APPEND_ITEM CCSDSVER      3 UINT   "CCSDS packet version number (See CCSDS 133.0-B-1)"
  APPEND_ITEM CCSDSTYPE     1 UINT   "CCSDS packet type (command or telemetry)"
    STATE TLM 0
    STATE CMD 1
  APPEND_ITEM CCSDSSHF      1 UINT   "CCSDS secondary header flag"
    STATE FALSE 0
    STATE TRUE 1
  APPEND_ID_ITEM CCSDSAPID 11 UINT 133 "CCSDS application process id"
  APPEND_ITEM CCSDSSEQFLAGS 2 UINT   "CCSDS sequence flags"
    STATE FIRST 1
    STATE CONT 0
    STATE LAST 2
    STATE NOGROUP 3
  APPEND_ITEM CCSDSSEQCNT  14 UINT   "CCSDS packet sequence count"
    OVERFLOW TRUNCATE
  APPEND_ITEM CCSDSLENGTH  16 UINT   "CCSDS packet data length"
  APPEND_ITEM XML 0 BLOCK "GPS Sensor XML Data"
    READ_CONVERSION hash_read_conversion.py TLM ZENDIR GPS_SENSOR_XML
  ITEM TIME 0 0 DERIVED "GPS sensor timestamp (seconds)"
    READ_CONVERSION openc3/conversions/processor_conversion.py XMLPROCESSOR TIME
  ITEM UTC 0 0 DERIVED "UTC timestamp of the GPS data"
    READ_CONVERSION openc3/conversions/processor_conversion.py XMLPROCESSOR UTC
  ITEM LATITUDE 0 0 DERIVED "Latitude (deg)"
    READ_CONVERSION openc3/conversions/processor_conversion.py XMLPROCESSOR LATITUDE
  ITEM LONGITUDE 0 0 DERIVED "Longitude (deg)"
    READ_CONVERSION openc3/conversions/processor_conversion.py XMLPROCESSOR LONGITUDE
  ITEM ALTITUDE 0 0 DERIVED "Altitude (m)"
    READ_CONVERSION openc3/conversions/processor_conversion.py XMLPROCESSOR ALTITUDE
  PROCESSOR XMLPROCESSOR hash_processor.py  XML
